FROM Expediente e
WHERE EXISTS (
    SELECT 1
    FROM ExpedienteDocumento d
    WHERE d.IdExpediente = e.IdExpediente
      AND EXISTS (
          SELECT 1
          FROM ExpedienteDocumentoOrigen o
          WHERE o.IdExpedienteDocumento = d.IdExpedienteDocumento
            AND EXISTS (
                SELECT 1
                FROM ExpedienteDocumentoOrigenDestino t
                WHERE t.IdExpedienteDocumentoOrigen = o.IdExpedienteDocumentoOrigen
                  AND t.IdPersonaDestino = @persona
            )
      )
)

COMBINACION (PRO) LA MEJOR
======================================
Filtro mas extraccion controlada... esto evita la expansion

SELECT e.IdExpediente, t.IdPersonaDestino
FROM Expediente e
WHERE EXISTS (
    SELECT 1
    FROM ExpedienteDocumentoOrigenDestino t
    JOIN ExpedienteDocumentoOrigen o 
        ON o.IdExpedienteDocumentoOrigen = t.IdExpedienteDocumentoOrigen
    JOIN ExpedienteDocumento d 
        ON d.IdExpedienteDocumento = o.IdExpedienteDocumento
    WHERE t.IdPersonaDestino = @persona
	AND d.IdExpediente = e.IdExpediente
)

OUTER APPLY (
    SELECT TOP 1 t.*
    FROM ExpedienteDocumentoOrigenDestino t
    JOIN ExpedienteDocumentoOrigen o 
        ON o.IdExpedienteDocumentoOrigen = t.IdExpedienteDocumentoOrigen
    JOIN ExpedienteDocumento d 
        ON d.IdExpedienteDocumento = o.IdExpedienteDocumento
    WHERE t.IdPersonaDestino = @persona
      AND d.IdExpediente = e.IdExpediente
    ORDER BY t.FechaCreacionAuditoria DESC
) t
